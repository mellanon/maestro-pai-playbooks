  FEATURE_ID=$(grep "Feature ID:" .maestro/CURRENT_FEATURE.md 2>/dev/null | awk -F: '{print $2}' | tr -d ' ')

    echo "PHASE_GUARD: SKIP - Feature already past specify phase (current: $PHASE)"
    exit 0
  fi
